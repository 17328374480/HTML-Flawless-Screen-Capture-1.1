<script>
(function() {
    // ==================== DOM 元素 ====================
    const startBtn = document.getElementById('startBtn');
    const stopBtn = document.getElementById('stopBtn');
    const timerDiv = document.getElementById('timerDisplay');
    const segmentDiv = document.getElementById('segmentDisplay');
    const bitrateSlider = document.getElementById('bitrateSlider');
    const bitrateDisplay = document.getElementById('bitrateDisplay');
    const codecSelect = document.getElementById('codecSelect');
    const chipContainer = document.getElementById('selectedCodecChip');
    const resWarning = document.getElementById('resWarning');

    const resRadios = document.querySelectorAll('input[name="res"]');
    const fpsRadios = document.querySelectorAll('input[name="fps"]');
    const chromaRadios = document.querySelectorAll('input[name="chroma"]');

    const audioEnhanceSelect = document.getElementById('audioEnhance');
    const gainSlider = document.getElementById('gainSlider');
    const gainSpan = document.getElementById('gainValue');
    const audioStatusDiv = document.getElementById('audioStatus');

    // ==================== 配置常量 ====================
    const SEGMENT_DURATION_MS = 2 * 60 * 60 * 1000; // 2小时 (毫秒)
    const TIMESLICE_INTERVAL = 1000; // 每秒触发一次数据缓存，防止内存溢出

    // ==================== 状态变量 ====================
    let rawStream = null;
    let audioContext = null;
    let processedStreamForRecord = null;
    let mediaRecorder = null;
    let recordedChunks = [];
    
    let startTime = null;
    let timerInterval = null;
    let recordingActive = false;
    
    let segmentIndex = 1; // 当前分段索引
    let segmentStartTime = null; // 当前分段开始时间
    let autoStopTimeout = null; // 自动分段定时器

    // ==================== 初始化与UI更新 ====================
    gainSlider.addEventListener('input', () => {
        let val = gainSlider.value;
        gainSpan.innerText = (val > 0 ? '+' : '') + val + ' dB';
        audioStatusDiv.innerHTML = val > 12 ? '⚠️ 增益过高' : '🎧 BBE+限制器 已优化';
    });

    function updateCodecChip() {
        const selected = codecSelect.options[codecSelect.selectedIndex].text;
        chipContainer.innerHTML = `<span class="chip" style="background:#2f4780; border-color:#98b4ff;">🔷 ${selected}</span>`;
    }
    codecSelect.addEventListener('change', updateCodecChip);
    updateCodecChip();

    bitrateSlider.addEventListener('input', (e) => {
        bitrateDisplay.innerText = e.target.value + ' Mbps';
    });

    // ==================== 辅助函数 ====================
    function getSelectedResolution() {
        for (let r of resRadios) if (r.checked) {
            const [w, h] = r.value.split('x').map(Number);
            return { width: w, height: h };
        }
        return { width: 3840, height: 2160 };
    }
    
    function getSelectedFPS() {
        for (let f of fpsRadios) if (f.checked) return parseInt(f.value, 10);
        return 60;
    }
    
    function getSelectedChroma() {
        for (let c of chromaRadios) if (c.checked) return c.value;
        return '420';
    }
    
    function getSelectedBitrateMbps() {
        return parseInt(bitrateSlider.value, 10);
    }
    
    function getSelectedCodecKey() {
        return codecSelect.value;
    }

    function buildMimeFromSelection() {
        const codecKey = getSelectedCodecKey();
        const chroma = getSelectedChroma();
        let baseMime = 'video/mp4';
        let codecPart = '';

        switch (codecKey) {
            case 'vp9': baseMime = 'video/webm'; codecPart = 'vp9'; break;
            case 'vp8': baseMime = 'video/webm'; codecPart = 'vp8'; break;
            case 'h264': baseMime = 'video/mp4'; codecPart = 'avc1.640034'; break;
            case 'av1':
                baseMime = 'video/mp4';
                if (chroma === '444') codecPart = 'av01.0.15M.08.0.110.01.01.01.0';
                else if (chroma === '422') codecPart = 'av01.0.15M.08.0.110.01.01.00.0';
                else codecPart = 'av01.0.15M.08';
                break;
            case 'h265': baseMime = 'video/mp4'; codecPart = 'hev1.1.6.L153.B0'; break;
            default: baseMime = 'video/mp4';
        }
        if (codecPart) return `${baseMime}; codecs="${codecPart}"`;
        return baseMime;
    }

    function getSupportedMimeOrFallback() {
        const desired = buildMimeFromSelection();
        if (MediaRecorder.isTypeSupported(desired)) return desired;
        
        const base = desired.split(';')[0];
        if (MediaRecorder.isTypeSupported(base)) {
            alert(`编码降级提示：当前浏览器不支持所选编码参数，已切换为基础容器`);
            return base;
        }
        return 'video/webm'; // 最终回退
    }

    // ==================== 获取屏幕流 ====================
    async function getCustomStream() {
        const { width, height } = getSelectedResolution();
        const fps = getSelectedFPS();
        
        const constraints = {
            video: { width: { ideal: width }, height: { ideal: height }, frameRate: { ideal: fps, max: fps } },
            audio: { echoCancellation: false, noiseSuppression: false, autoGainControl: false }
        };

        try {
            const displayStream = await navigator.mediaDevices.getDisplayMedia(constraints);
            const track = displayStream.getVideoTracks()[0];
            const settings = track.getSettings();
            
            // 更新UI提示实际分辨率
            resWarning.innerHTML = `✅ 实际: ${settings.width}x${settings.height} @ ${settings.frameRate}FPS`;
            resWarning.style.color = '#66ff99';
            
            return displayStream;
        } catch (err) {
            alert('获取屏幕失败: ' + err.message);
            throw err;
        }
    }

    // ==================== 计时器逻辑 ====================
    function formatTime(ms) {
        let totalSeconds = Math.floor(ms / 1000);
        let h = Math.floor(totalSeconds / 3600);
        let m = Math.floor((totalSeconds % 3600) / 60);
        let s = totalSeconds % 60;
        return `${h.toString().padStart(2,'0')}:${m.toString().padStart(2,'0')}:${s.toString().padStart(2,'0')}`;
    }
    
    function startTimer() {
        segmentStartTime = Date.now();
        timerInterval = setInterval(() => {
            const elapsed = Date.now() - segmentStartTime;
            timerDiv.innerText = formatTime(elapsed);
        }, 1000);
    }
    
    function resetTimerUI() {
        if(timerInterval) clearInterval(timerInterval);
        timerDiv.innerText = "00:00:00";
    }

    // ==================== 录制核心：分段保存逻辑 ====================
    
    // 触发下载并保存当前段
    function saveSegment() {
        if (recordedChunks.length === 0) return;

        const mime = mediaRecorder ? mediaRecorder.mimeType : 'video/mp4';
        const blob = new Blob(recordedChunks, { type: mime });
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;

        // 动态生成文件名
        const { width, height } = getSelectedResolution();
        const fps = getSelectedFPS();
        const bitrate = getSelectedBitrateMbps();
        const codecName = codecSelect.options[codecSelect.selectedIndex].text.replace(/[ .()]/g, '');
        const enhanceTag = audioEnhanceSelect.value === 'bbe' ? '_BBE' : '';
        const now = new Date();
        const ts = `${now.getFullYear()}-${now.getMonth()+1}-${now.getDate()}_${now.getHours()}-${now.getMinutes()}`;
        
        // 关键：文件名包含PartX
        a.download = `${codecName}_${width}x${height}_Part${segmentIndex}_${ts}${enhanceTag}.mp4`;
        
        document.body.appendChild(a); // Firefox 需要
        a.click();
        document.body.removeChild(a);
        
        setTimeout(() => URL.revokeObjectURL(url), 1000);
        
        // 清空缓存，准备下一段
        recordedChunks = [];
        console.log(`✅ 分段 ${segmentIndex} 已保存`);
    }

    // 切换到下一个分段
    function switchToNextSegment() {
        // 1. 停止当前录制器，触发onstop -> saveSegment
        if (mediaRecorder && mediaRecorder.state === 'recording') {
            mediaRecorder.stop(); 
        }
        
        // 2. 索引递增
        segmentIndex++;
        segmentDiv.innerText = `Part ${segmentIndex}`;
        console.log(`🔄 切换至分段 ${segmentIndex}`);
        
        // 3. 立即启动新的录制器 (在onstop回调中执行，确保安全)
    }

    // 录制停止回调 (无论是手动停止还是自动分段)
    function onRecorderStop() {
        saveSegment();
        
        // 如果录制仍在激活状态，说明是自动分段触发的，需要立即启动下一段
        if (recordingActive) {
            startNewRecorderInstance();
        } else {
            // 彻底停止
            cleanupAll();
        }
    }

    // 创建新的录制器实例
    function startNewRecorderInstance() {
        if (!processedStreamForRecord) {
            console.error('流已断开，无法开始新分段');
            stopAll();
            return;
        }

        const mimeType = getSupportedMimeOrFallback();
        const bitrateBps = getSelectedBitrateMbps() * 1_000_000;
        
        const options = { 
            mimeType: mimeType, 
            videoBitsPerSecond: bitrateBps 
        };

        try {
            mediaRecorder = new MediaRecorder(processedStreamForRecord, options);
        } catch (e) {
            alert('创建录制器失败: ' + e.message);
            stopAll();
            return;
        }

        mediaRecorder.ondataavailable = (e) => {
            if (e.data.size > 0) recordedChunks.push(e.data);
        };
        
        mediaRecorder.onstop = onRecorderStop;
        
        mediaRecorder.onerror = (e) => {
            console.error('录制错误:', e);
            // 遇到错误尝试保存当前数据并停止
            recordingActive = false; 
            mediaRecorder.stop();
        };

        // 关键：启动录制，并设置timeslice，防止内存堆积
        mediaRecorder.start(TIMESLICE_INTERVAL);
        
        // 重置计时器
        resetTimerUI();
        startTimer();

        // 设置2小时后的自动分段触发器
        clearTimeout(autoStopTimeout);
        autoStopTimeout = setTimeout(() => {
            if (recordingActive) switchToNextSegment();
        }, SEGMENT_DURATION_MS);
    }

    // ==================== 彻底停止与清理 ====================
    function stopAll() {
        recordingActive = false;
        clearTimeout(autoStopTimeout);
        
        if (mediaRecorder && mediaRecorder.state !== 'inactive') {
            mediaRecorder.stop(); // 这会触发 onRecorderStop -> saveSegment
        } else {
            cleanupAll();
        }
        
        startBtn.disabled = false;
        stopBtn.disabled = true;
        segmentDiv.innerText = '已停止';
    }

    function cleanupAll() {
        if (audioContext && audioContext.state !== 'closed') {
            audioContext.close();
        }
        if (rawStream) {
            rawStream.getTracks().forEach(t => t.stop());
        }
        rawStream = null;
        audioContext = null;
        processedStreamForRecord = null;
        resetTimerUI();
    }

    // ==================== 开始录制入口 ====================
    async function startRecording() {
        try {
            // 重置状态
            segmentIndex = 1;
            segmentDiv.innerText = `Part 1`;
            recordedChunks = [];
            recordingActive = true;

            // 1. 获取流
            rawStream = await getCustomStream();
            
            // 2. 音频处理
            const audioTracks = rawStream.getAudioTracks();
            if (audioTracks.length === 0) {
                audioStatusDiv.innerHTML = '🔇 无音频';
            } else {
                // BBE处理逻辑 (简化版，保持逻辑一致)
                if (audioEnhanceSelect.value === 'bbe') {
                    try {
                        audioContext = new (window.AudioContext || window.webkitAudioContext)({ sampleRate: 48000 });
                        const source = audioContext.createMediaStreamSource(new MediaStream([audioTracks[0]]));
                        
                        // 增益与限制器
                        const gainNode = audioContext.createGain();
                        gainNode.gain.value = Math.pow(10, parseFloat(gainSlider.value) / 20);
                        
                        const limiter = audioContext.createDynamicsCompressor();
                        limiter.threshold.value = -2;
                        limiter.knee.value = 0;
                        limiter.ratio.value = 20;
                        limiter.attack.value = 0.001;
                        limiter.release.value = 0.01;

                        // BBE滤波器
                        const lowComp = audioContext.createBiquadFilter();
                        lowComp.type = 'peaking';
                        lowComp.frequency.value = 80;
                        lowComp.Q.value = 1.5;
                        lowComp.gain.value = 2;

                        const highExciter = audioContext.createBiquadFilter();
                        highExciter.type = 'highshelf';
                        highExciter.frequency.value = 5000;
                        highExciter.gain.value = 4;

                        source.connect(lowComp);
                        lowComp.connect(highExciter);
                        highExciter.connect(gainNode);
                        gainNode.connect(limiter);
                        
                        const dest = audioContext.createMediaStreamDestination();
                        limiter.connect(dest);
                        
                        // 合并视频轨和处理后的音频轨
                        processedStreamForRecord = new MediaStream([
                            rawStream.getVideoTracks()[0],
                            dest.stream.getAudioTracks()[0]
                        ]);
                        audioStatusDiv.innerHTML = '🎧 BBE处理中';

                    } catch(e) {
                        console.warn("音频处理失败，使用原始流", e);
                        processedStreamForRecord = rawStream;
                    }
                } else {
                    processedStreamForRecord = rawStream;
                }
            }

            // 3. 启动第一个分段
            startNewRecorderInstance();

            startBtn.disabled = true;
            stopBtn.disabled = false;

            // 监听用户手动停止共享
            rawStream.getVideoTracks()[0].onended = () => {
                if (recordingActive) stopAll();
            };

        } catch (err) {
            console.error('启动录制失败:', err);
            cleanupAll();
            startBtn.disabled = false;
            stopBtn.disabled = true;
        }
    }

    // 事件绑定
    startBtn.addEventListener('click', startRecording);
    stopBtn.addEventListener('click', stopAll);

    // 美化radio
    document.querySelectorAll('.radio-item').forEach(item => {
        const radio = item.querySelector('input[type="radio"]');
        if (radio && radio.checked) item.classList.add('selected');
        radio?.addEventListener('change', function(e) {
            document.querySelectorAll(`input[name="${this.name}"]`).forEach(r => {
                r.closest('.radio-item').classList.remove('selected');
            });
            if (this.checked) this.closest('.radio-item').classList.add('selected');
        });
    });

    // 页面关闭时尝试保存
    window.addEventListener('beforeunload', (e) => {
        if (recordingActive) {
            e.preventDefault();
            e.returnValue = '';
            stopAll();
        }
    });
})();
